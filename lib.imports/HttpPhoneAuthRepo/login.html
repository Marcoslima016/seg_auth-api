@override
Future login({LoginAttemptCredentials credentials, UserModel userModel}) async {
  Map dataJson = {
    "usuario": credentials.user,
    "senha": credentials.pass,
    "token_push": SegAppConfig.instance.appController.deviceInfo.tokenPush,
    "so": SegAppConfig.instance.appController.deviceInfo.platInfo.so,
    "so_ver": SegAppConfig.instance.appController.deviceInfo.platInfo.version,
    "app_id": 2,
  };

  HttpRequestResult requestResult = await _client.post(dataJson: dataJson, enableJwt: false, endPoint: "login.php");

  ///=========== SUCESSO ===========
  ///
  if (requestResult.status == ResultStatus.sucess) {
    //
    Response result = requestResult.result;
    Map data = result.data["data"];

    // LoginRequestResult requestResultModel = LoginRequestResult(
    //   nome: data["info"],
    //   token: data["token"],
    // );

    UserModel user = await userModel.convertFromDatabase(data);
    user.pass = credentials.pass;
    user.user = credentials.user;

    return user;
  }

  ///============= ERRO ============
  ///
  if (requestResult.status == ResultStatus.error || requestResult.status == ResultStatus.connectionFail) {
    throw (requestResult.result);
  }
}
