Future request({@required ClientHttp client, String endPoint, Map dataJson, bool enableJwt, Future Function(Response response) onSuccess}) async {
  HttpRequestResult requestResult = await client.post(dataJson: dataJson, enableJwt: enableJwt, endPoint: endPoint);

  ///=========== SUCESSO ===========
  ///
  if (requestResult.status == ResultStatus.sucess) {
    return await onSuccess(requestResult.result);
  }

  ///============= ERRO ============
  ///
  if (requestResult.status == ResultStatus.error || requestResult.status == ResultStatus.connectionFail) {
    throw (requestResult.result);
  }
}
